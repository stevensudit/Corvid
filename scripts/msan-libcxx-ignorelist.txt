# Sanitizer ignorelist for MSAN builds.
#
# Passed via -fsanitize-ignorelist=... to both the libc++ build
# (scripts/build_msan_libcxx.sh) and the project build (tests/CMakeLists.txt
# when SANITIZER=msan).
#
# Currently empty: with libc++ correctly instrumented (the libunwind PUBLIC
# -> PRIVATE patch in build_msan_libcxx.sh stops `-fno-sanitize=memory` from
# leaking into libc++.so), no libc++ false positives need suppressing.
# Leaving this file in place as scaffolding so future false positives can be
# suppressed surgically without re-wiring the cmake plumbing.
#
# Pattern semantics:
#   src: -- matches the file containing the function definition (header path
#           for templates, source path for ordinary functions). Suppressing a
#           file also stops shadow tracking of stores within it, so go narrow.
#   fun: -- matches the demangled function name; fallback for templates whose
#           IR debug info points to the instantiation site.

[memory]
